# 클플 프록시 가이드

로컬 프록시 대신 **Cloudflare Workers 기반 GitHub Copilot CORS Proxy**를 쓰고 싶은 사람이,
자기 Cloudflare 계정에 **그대로 복붙해서 직접 배포**할 수 있게 넣어둔 문서다.

즉 이 문서는:

- 공용 프록시 운영 안내가 아니라
- **각자 자기 계정에 직접 배포해서 쓰는 용도**이고
- 아래 코드를 **복붙해서 Worker에 넣으라는 목적**이다.

---

## 지원 엔드포인트

- `/chat/completions`
- `/v1/chat/completions`
- `/v1/messages`
- `/responses`
- `/v1/responses`

---

## 배포 방법

1. Cloudflare Dashboard로 이동
2. **Workers & Pages** → **Create Worker**
3. 아래 코드 전체를 **그대로 복붙**하고 Deploy
4. 커스텀 모델 Proxy URL에 Worker URL 입력
   - 예: `https://my-proxy.username.workers.dev`

요청 시 `Authorization: Bearer <github_oauth_token>` 또는 `x-api-key` 헤더로 GitHub OAuth 토큰을 보내면,
Worker가 `tid` 토큰을 자동 발급받아 Copilot API로 프록시한다.

---

## Worker 코드

```js
// ══════════════════════════════════════════════════════════
// GitHub Copilot CORS Proxy — Cloudflare Workers
// ══════════════════════════════════════════════════════════
// 지원 엔드포인트: /chat/completions, /v1/messages, /responses
//
// 사용법:
//   1. Cloudflare Dashboard → Workers & Pages → Create Worker
//   2. 이 코드 전체를 붙여넣기 → Deploy
//   3. 커스텀 모델 Proxy URL에 워커 URL 입력
//      예: https://my-proxy.username.workers.dev
//
// 요청 시 Authorization 헤더에 GitHub OAuth 토큰을 넣으면
// 이 워커가 자동으로 tid 토큰을 발급받아 Copilot API로 프록시합니다.
// ══════════════════════════════════════════════════════════

const COPILOT_TOKEN_URL = "https://api.github.com/copilot_internal/v2/token";
const COPILOT_API_BASE = "https://api.githubcopilot.com";
const CHAT_VERSION = "0.26.2025030601";
const CODE_VERSION = "1.99.2025030601";
const USER_AGENT = `GitHubCopilotChat/${CHAT_VERSION}`;

// 지원하는 경로 목록
const ALLOWED_PATHS = [
  "/chat/completions",
  "/v1/chat/completions",
  "/v1/messages",
  "/responses",
  "/v1/responses",
];

function generateHexId(length = 64) {
  const arr = new Uint8Array(length / 2);
  crypto.getRandomValues(arr);
  return Array.from(arr, (b) => b.toString(16).padStart(2, "0")).join("");
}

// ── tid 토큰 발급 ──
async function refreshTidToken(apiKey) {
  const resp = await fetch(COPILOT_TOKEN_URL, {
    method: "GET",
    headers: {
      Accept: "application/json",
      Authorization: `Bearer ${apiKey}`,
      "User-Agent": USER_AGENT,
    },
  });

  if (!resp.ok) {
    const text = await resp.text();
    throw new Error(`tid token failed (${resp.status}): ${text}`);
  }

  const data = await resp.json();
  if (!data.token || !data.expires_at) {
    throw new Error("tid token response missing token or expires_at");
  }

  return data.token;
}

// ── CORS 헤더 ──
const CORS_HEADERS = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Methods": "GET, POST, PUT, DELETE, OPTIONS",
  "Access-Control-Allow-Headers": "*",
  "Access-Control-Max-Age": "86400",
};

// ── 경로 정규화: /v1/chat/completions → /chat/completions 등 ──
function normalizePath(pathname) {
  // /v1/chat/completions → /chat/completions
  if (pathname === "/v1/chat/completions") return "/chat/completions";
  // /v1/responses → /responses
  if (pathname === "/v1/responses") return "/responses";
  return pathname;
}

// ── 메인 핸들러 ──
export default {
  async fetch(request) {
    // preflight
    if (request.method === "OPTIONS") {
      return new Response(null, { status: 204, headers: CORS_HEADERS });
    }

    const url = new URL(request.url);

    // health check
    if (url.pathname === "/" || url.pathname === "/health") {
      return new Response(
        JSON.stringify({
          status: "ok",
          endpoints: ALLOWED_PATHS,
          usage: "Authorization: Bearer <github_oauth_token>",
        }),
        {
          headers: { "Content-Type": "application/json", ...CORS_HEADERS },
        }
      );
    }

    // 경로 확인
    const normalizedPath = normalizePath(url.pathname);
    if (!ALLOWED_PATHS.includes(normalizedPath) && !ALLOWED_PATHS.includes(url.pathname)) {
      return new Response(
        JSON.stringify({
          error: "Not Found",
          allowed: ALLOWED_PATHS,
        }),
        {
          status: 404,
          headers: { "Content-Type": "application/json", ...CORS_HEADERS },
        }
      );
    }

    if (request.method !== "POST") {
      return new Response(
        JSON.stringify({ error: "Method Not Allowed" }),
        {
          status: 405,
          headers: { "Content-Type": "application/json", ...CORS_HEADERS },
        }
      );
    }

    try {
      // ── API 키 추출 (Authorization 또는 x-api-key) ──
      const authHeader = request.headers.get("Authorization") || "";
      const xApiKey = request.headers.get("x-api-key") || "";
      const apiKey = authHeader.startsWith("Bearer ")
        ? authHeader.slice(7)
        : xApiKey;

      if (!apiKey) {
        return new Response(
          JSON.stringify({ error: "API key required (Authorization: Bearer <token> or x-api-key header)" }),
          {
            status: 401,
            headers: { "Content-Type": "application/json", ...CORS_HEADERS },
          }
        );
      }

      // ── tid 토큰 발급 ──
      let tidToken;
      try {
        tidToken = await refreshTidToken(apiKey);
      } catch (e) {
        return new Response(
          JSON.stringify({ error: e.message }),
          {
            status: 401,
            headers: { "Content-Type": "application/json", ...CORS_HEADERS },
          }
        );
      }

      // ── 요청 본문 ──
      const rawBody = await request.text();
      const hasVision =
        rawBody.includes('"image_url"') ||
        rawBody.includes('"image"') ||
        rawBody.includes('"input_image"');
      const isStream =
        rawBody.includes('"stream":true') ||
        rawBody.includes('"stream": true');

      // ── UUID 생성 ──
      const sessionId = crypto.randomUUID() + Date.now().toString();
      const interactionId = crypto.randomUUID();
      const requestId = crypto.randomUUID();

      // ── Copilot 대상 URL 결정 ──
      const targetPath = normalizePath(url.pathname);
      const copilotUrl = `${COPILOT_API_BASE}${targetPath}`;

      // ── Copilot 공통 헤더 ──
      const copilotHeaders = {
        Authorization: `Bearer ${tidToken}`,
        "Content-Type": "application/json",
        "Copilot-Integration-Id": "vscode-chat",
        "Editor-plugin-version": `copilot-chat/${CHAT_VERSION}`,
        "Editor-version": `vscode/${CODE_VERSION}`,
        "User-Agent": USER_AGENT,
        "Vscode-Machineid": generateHexId(64),
        "Vscode-Sessionid": sessionId,
        "X-Github-Api-Version": "2025-10-01",
        "X-Initiator": "user",
        "X-Interaction-Id": interactionId,
        "X-Interaction-Type": "conversation-panel",
        "X-Request-Id": requestId,
        "X-Vscode-User-Agent-Library-Version": "electron-fetch",
      };

      // /v1/messages (Anthropic) → anthropic-version 헤더 추가
      if (targetPath === "/v1/messages") {
        copilotHeaders["anthropic-version"] = "2023-06-01";
        // Anthropic 요청에서 클라이언트가 보낸 anthropic-version이 있으면 그걸 사용
        const clientAnthropicVersion = request.headers.get("anthropic-version");
        if (clientAnthropicVersion) {
          copilotHeaders["anthropic-version"] = clientAnthropicVersion;
        }
      }

      // 비전 요청
      if (hasVision) {
        copilotHeaders["Copilot-Vision-Request"] = "true";
      }

      // ── Copilot API 호출 ──
      const copilotResp = await fetch(copilotUrl, {
        method: "POST",
        headers: copilotHeaders,
        body: rawBody,
      });

      // ── 응답 패스스루 + CORS 헤더 추가 ──
      const responseHeaders = {
        "Content-Type":
          copilotResp.headers.get("Content-Type") || "application/json",
        ...CORS_HEADERS,
      };

      if (isStream) {
        responseHeaders["Cache-Control"] = "no-cache";
        responseHeaders["Connection"] = "keep-alive";
      }

      return new Response(copilotResp.body, {
        status: copilotResp.status,
        headers: responseHeaders,
      });
    } catch (e) {
      return new Response(
        JSON.stringify({ error: e.message }),
        {
          status: 500,
          headers: { "Content-Type": "application/json", ...CORS_HEADERS },
        }
      );
    }
  },
};
```
